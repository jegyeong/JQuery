수행내용

dom객체
var domele = document.getElementsByTagName('h1')[0];
dom객체변수는 스크립트의 문법에 의한다
domele.style.color = 'red';
css()는 jQuery의 메소드 이기 때문에 사용 할 수 없다
domele.css('color', 'red'); -->>오류
dom 객체로 jQuery메소드 를 사용하기 위해서는
dom객체를 $()로 감싸준다
$(domele).css('color', 'blue');

jQuery객체
var jq = $('h2');
jq.css('color', 'red'); 메소드 체이닝
$('h2').css('color', 'orange');
