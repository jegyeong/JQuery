$(“html”, {propertis})
인수로 전달된 html문자열로 새로운 엘리먼트를 직접 생성한다
생성된 엘리먼트를 DOM트리의 원하는 부분에 삽입하여 실행중에 문서를 만들 수 있다
propertis 는새로만들어진 요소의 속성, 이벤트, 함수 등을 지정

이미지 추가 버튼을 클릭하면 이미지를 생성해서 body안에 이미지를 나타나게 한다
이미지는 배열로 생성해놓고 random을 발생시켜 이미지를 나타나게 한다

imgarr = ['Chrysanthemum.jpg','Desert.jpg','Hydrangeas.jpg']
$('button').click(function(){
   rand = parseInt(Math.random()*imgarr.length);
  //img태그를 만든다 , 속성은 src, width, height
    $('<img/>', {
     'src' : '../imges/' + imgarr[rand],
     'width' : '150px',
     'height' : '150px'
    }).appendTo('body');
})
