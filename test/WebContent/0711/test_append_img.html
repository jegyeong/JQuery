내부에 추가하는 메소드
----뒤쪽으로 추가 ------------------------
선택자.append('내용') ==== 선택자 안쪽
('내용').appendTo('선택자') == 선택자 안쪽

선택자.after('내용') ==선택자 밖
('내용').insertAfter('선택자') ==선택자 밖

-----앞쪽으로------------------
선택자.prepend('내용') ==== 선택자 안쪽
('내용').prependTo('선택자') == 선택자 안쪽

선택자.before('내용') ==선택자 밖
('내용').insertBefore('선택자') ==선택자 밖 




     $('#start').click(function() {
