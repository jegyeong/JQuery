복사
clone() : 요소만 복사하기
clone(true) : 요소와 이벤트 복사하기 

  //p의 이벤트 
    $('p').click(function(){
		 $(this).css('font-size', '+=2px');
	})

  //clone버튼 클릭  - 요소만 복사 
	$('#btn1').click(function(){
		 //p를 복사해서 body에 append , appendTo로  추가 
		 //$('p').clone().appendTo('body')
		 $('body').append($('p').clone());
	})
  //clone(true)버튼 클릭  - 요소와 이빈트복사 
	$('#btn2').click(function(){
		 $('body').append($('p').clone(true));
	})
