선택자.wrap(html) : 선택자 각각을 html로 감싼다
선택자.wrapAll(html) : 선택자를 한꺼번에 html로 감싼다
unwrap() : wrap를 해제 힌다 
	
  $("#btn1").click(function(){
       //$("p").wrap("<div></div>");
       $("p").wrapAll("<div></div>"); 
    });
    
    $("#btn2").click(function(){
          $("p").unwrap();  
    });
